 class TrainRepositoryTest{
 /**
  * Verifies that loadAll handles IO errors (e.g. when the "file path"
  * actually points to a directory) by catching the exception and
  * returning an empty list instead of throwing.
  */
 @Test
 void testLoadAllHandlesIOExceptionGracefully(@TempDir Path tempDir) {
     // Create a directory and use its path as the "file"
     File dirAsFile = tempDir.resolve("train_dir").toFile();
     assertTrue(dirAsFile.mkdir());
 
     // When TrainRepository tries to open this as a file, FileReader will
     // throw an IOException which should be caught inside loadAll().
     TrainRepository repo = new TrainRepository(dirAsFile.getAbsolutePath());
     List<Train> trains = repo.loadAll();
 
     // Method should not throw and should simply return an empty list
     assertNotNull(trains);
     assertTrue(trains.isEmpty());
 }
 
 /**
  * Verifies that saveAll catches IOExceptions (for example when trying
  * to write to a directory path) and does not propagate them to callers.
  */
 @Test
 void testSaveAllHandlesIOExceptionGracefully(@TempDir Path tempDir) {
     // Again, use a directory instead of a regular file
     File dirAsFile = tempDir.resolve("train_dir").toFile();
     assertTrue(dirAsFile.mkdir());
 
     TrainRepository repo = new TrainRepository(dirAsFile.getAbsolutePath());
 
     List<Train> trains = List.of(
             new Train(
                     "T999",
                     "TestDest",
                     LocalDate.of(2035, 5, 5),
                     LocalTime.of(15, 30),
                     50,
                     10,
                     40.0,
                     100.0,
                     TrainStatus.ACTIVE
             )
     );
 
     // saveAll should catch the IOException internally and not throw
     assertDoesNotThrow(() -> repo.saveAll(trains));
 
     // Path should still be a directory (no file created with that name)
     assertTrue(dirAsFile.isDirectory());
 }
