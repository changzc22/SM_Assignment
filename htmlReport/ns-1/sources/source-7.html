 package test;
 
 import oopt.assignment.model.IStaffRepository;
 import oopt.assignment.model.Staff;
 import oopt.assignment.service.StaffService;
 import oopt.assignment.ui.StaffUI;
 import oopt.assignment.util.PasswordUtil;
 import org.junit.jupiter.api.DisplayName;
 import org.junit.jupiter.api.Test;
 
 import java.io.ByteArrayInputStream;
 import java.io.InputStream;
 import java.util.Collection;
 import java.util.LinkedHashMap;
 
 import static org.junit.jupiter.api.Assertions.*;
 
 @DisplayName("Staff UI Interaction Tests")
 public class StaffUITest {
 
     /**
      * tests the login interaction flow via the console.
      * It simulates a user typing an ID and Password by injecting a byte stream
      * into System.in, then verifies that the UI returns the correct logged-in ID.
      */
     @Test
     void testLoginInputFlow() {
         // 1. Mock Service with a known user
         StaffService mockService = new StaffService(new MockRepo());
 
         // 2. Simulate User Input: "S001" [Enter] "Pass123" [Enter]
         // We use ByteArrayInputStream to "fake" typing on the keyboard
         String simulatedInput = "S001\nPass123\n";
         InputStream in = new ByteArrayInputStream(simulatedInput.getBytes());
         System.setIn(in); // Redirect System.in to read from our simulated input
 
         // 3. Run the static login method
         // The UI should read "S001" and "Pass123", validate them, and return "S001"
         String resultId = StaffUI.handleLogin(mockService);
 
         assertEquals("S001", resultId);
     }
 
     // Simple mock repository to provide known data for the UI test
     static class MockRepo implements IStaffRepository {
         public LinkedHashMap<String, Staff> getAll() {
             LinkedHashMap<String, Staff> map = new LinkedHashMap<>();
             // Create a user with a known password hash
             String hash = PasswordUtil.hashPassword("Pass123");
             map.put("S001", new Staff("Test", "012", "999", "S001", hash, 0));
             return map;
         }
         public void saveAll(Collection<Staff> staffList) {}
     }
 }
